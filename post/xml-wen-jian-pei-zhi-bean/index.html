XML文件配置IOC容器

1、相关依赖

spring-context

<dependency>
    <groupId>org.springframework</groupId>
    <artifactId>spring-context</artifactId>
    <version>5.2.12.RELEASE</version>
</dependency>

2、编写xml配置文件

2.1、初始配置文件

<beans xmlns="http://www.springframework.org/schema/beans"
       xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
       xmlns:context="http://www.springframework.org/schema/context"
       xsi:schemaLocation="http://www.springframework.org/schema/beans
http://www.springframework.org/schema/beans/spring-beans.xsd
http://www.springframework.org/schema/context
http://www.springframework.org/schema/context/spring-context.xsd">

</beans>

2.2、配置bean

创建bean

1、常规bean
<!-- id：唯一标识       class：全路径类名-->
<bean id="" class=""/>

2、匿名bean
<!-- id可以不写，匿名bean默认id格式为 类名#序号  序号从0开始-->
<!-- 类名：xx  默认id：xx#0      如果有多个匿名类则按顺序编号 -->
<bean class="xx"/>

2.2.1、setter注入属性

<bean id="" class="">
    1、基本数据类型
    <property name="属性名" value="值"/>
    2、对象
    <property name="属性名" ref="要引用的bean id"/>
    3、集合
    3.1、List
    <property name="">
        <list>
            <value>值</value>
            ...
        </list>
    </property>

    3.2、Set
    <property name="">
        <set>
            <value>sad</value>
        </set>
    </property>

    3.3、Map
    <property name="">
        <map>
            <entry key="" value=""/>
            ...
        </map>
    </property>


    4、Properties
    <property name="">
        <props>
            <prop key="">值</prop>
            ...
        </props>
    </property>
</bean>

2.2.2、构造器注入属性

<bean>
    1、按构造器参数名
    <constructor-arg name="参数名1" value=""/>
    <constructor-arg name="参数名2" value=""/>
    ...

    2、按构造器参数位置
    <constructor-arg index="0" value=""/>
    <constructor-arg name="1" value=""/>
    ...

</bean>
